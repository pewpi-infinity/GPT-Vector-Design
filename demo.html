Use this code in other repositories to integrate with the token/wallet system: 

import { setupIntegration } from './pewpi-shared/integration-listener.js';
import { tokenService } from './pewpi-shared/token-service.js';

// Setup event listeners
const listener = setupIntegration({
  onTokenCreated: (token) => {
    console.log('New token:', token);
    updateYourAppState(token);
  },
  onLoginChanged: (data) => {
    console.log('Login changed:', data.loggedIn);
    updateAuthState(data.user);
  }
});

// Create tokens
await tokenService.createToken({
  type: 'bronze',
  value: 1,
  userId: getCurrentUserId()
});
